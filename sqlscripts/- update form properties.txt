CREATE TABLE REPORTCLASSPROPERTIES
( CLASS_ID NUMBER(36)
, PACTION NUMBER(36)
, XLSTEMPLATE VARCHAR2(30)
, XLSTOPLINE NUMBER(36)
, USEHEADER NUMBER(36)
, TXTTEMPLATE VARCHAR(30)
);

CREATE TABLE REFMAP
( REF_ID NUMBER(36)
, CLASS_ID NUMBER(36)
, REFNAME VARCHAR(50)
, REFLOOKUP VARCHAR(50)
);
CREATE SEQUENCE SEQ_REFMAP START WITH 1;

CREATE TABLE XLSMAP
( XMAP_ID NUMBER(36)
, CLASS_ID NUMBER(36)
, FIELDNUMBER NUMBER(36)
, FIELDNAME VARCHAR(100)
);
CREATE SEQUENCE SEQ_XLSMAP START WITH 1;

CREATE TABLE TXTMAP
( TMAP_ID NUMBER(36)
, CLASS_ID NUMBER(36)
, FIELDNUMBER NUMBER(36)
, FIELDLENGTH NUMBER(36)
, FIELDTYPECODE NUMBER(36)
);
CREATE SEQUENCE SEQ_TXTMAP START WITH 1;

CREATE TABLE ATTRLIST
( ALIST_ID NUMBER(36)
, CLASS_ID NUMBER(36)
, FIELDNAME VARCHAR2(100)
);
CREATE SEQUENCE SEQ_ATTRLIST START WITH 1;
